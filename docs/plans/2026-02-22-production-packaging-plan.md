### Task 2: Create the app icon generation script

**Files:**
- Create: `scripts/generate-app-icon.js`

This script takes the SVG source and generates:
- `assets/icon.png` (1024x1024 for electron-builder input)
- `assets/icon.icns` is auto-generated by electron-builder from the PNG

**Step 1: Create the icon generation script**

```js
#!/usr/bin/env node
/**
 * Generate app icon PNG (1024x1024) from SVG source.
 * electron-builder converts PNG → .icns automatically.
 * Run: node scripts/generate-app-icon.js
 */
const fs = require('fs');
const path = require('path');

async function generate() {
  let sharp;
  try {
    sharp = require('sharp');
  } catch {
    console.error('sharp not installed. Run: npm install -D sharp');
    process.exit(1);
  }

  const assetsDir = path.join(__dirname, '..', 'assets');
  const svgPath = path.join(assetsDir, 'app-icon.svg');

  if (!fs.existsSync(svgPath)) {
    console.error(`Missing: ${svgPath}`);
    console.error('Create assets/app-icon.svg first (1024x1024 viewBox recommended).');
    process.exit(1);
  }

  const svg = fs.readFileSync(svgPath);

  // 1024x1024 PNG — electron-builder converts this to .icns
  await sharp(svg)
    .resize(1024, 1024)
    .png()
    .toFile(path.join(assetsDir, 'icon.png'));
  console.log('Created icon.png (1024x1024)');

  console.log('\nDone! electron-builder will convert icon.png → icon.icns at build time.');
}

generate().catch(console.error);
```


### Task 5: Add the native addon rebuild script

**Files:**
- Create: `scripts/rebuild-native.js`
- Modify: `package.json`

**Step 1: Create the rebuild script**

This script rebuilds the native addon against the packaged Electron's Node ABI using @electron/rebuild:

```js
#!/usr/bin/env node
/**
 * Rebuild native addon against Electron's Node ABI.
 * Must run BEFORE electron-builder packages the app.
 */
const { rebuild } = require('@electron/rebuild');
const path = require('path');

const electronVersion = require('electron/package.json').version;

async function main() {
  console.log(`Rebuilding native addon for Electron ${electronVersion}...`);

  await rebuild({
    buildPath: path.join(__dirname, '..'),
    electronVersion,
    arch: process.arch,
    onlyModules: [], // rebuild all native modules
    force: true,
  });

  console.log('Native addon rebuilt successfully for Electron ABI.');
}

main().catch((err) => {
  console.error('Rebuild failed:', err);
  process.exit(1);
});
```

**Step 2: Update package.json scripts**

Add/modify these scripts in `package.json`:

```json
{
  "scripts": {
    "rebuild:electron": "node scripts/rebuild-native.js",
    "predist": "npm run build && npm run rebuild:electron",
    "dist": "electron-builder"
  }
}
```

The `predist` hook ensures everything is built and the native addon is recompiled before `electron-builder` runs.

**Step 3: Test the rebuild**

Run:
```bash
npm run rebuild:electron
```
Expected: `Rebuilding native addon for Electron 35.7.5...` followed by success message.

**Step 4: Verify dev mode still works after rebuild**

The rebuild changes the native addon to Electron ABI. Dev mode loads via Electron too, so it should work:

Run:
```bash

```
Expected: App starts normally with tray icon.
